16  package com.irurueta.reactorworkshop.arithmeticsequence.domain.services;
17  
18  import org.springframework.stereotype.Service;
19  
20  /**
21   * Exhaustively computes sum of elements in an arithmetic sequence.
22   * Because elements in the sequence are iterated to compute their sum, computation will take longer
23   * as the number of elements in the sequence increases.
24   */
25  @Service
26  public class ExhaustiveArithmeticSequenceService implements ArithmeticSequenceService {
27  
28      /**
29       * Computes sum of elements in an arithmetic sequence.
30       *
31       * @param minValue minimum value of first element in the sequence.
32       * @param step step between elements in the sequence.
33       * @param count number of elements in the sequence.
34       * @return sum of all elements in the sequence.
35       */
36      @Override
37      public int compute(final int minValue, final int step, final int count) {
38          int value = minValue;
39          int result = minValue;
40          for (int i = 1; i < count; i++) {
41              value += step;
42              result += value;
43          }
44  
45          return result;
46      }
47  }
