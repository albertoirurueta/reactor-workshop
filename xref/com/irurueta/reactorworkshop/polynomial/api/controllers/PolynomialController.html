16  package com.irurueta.reactorworkshop.polynomial.api.controllers;
17  
18  import com.irurueta.reactorworkshop.polynomial.api.dto.EvaluationStepsDto;
19  import com.irurueta.reactorworkshop.polynomial.api.dto.MultiplePolynomialEvaluationResultDto;
20  import com.irurueta.reactorworkshop.polynomial.api.dto.factories.MultiplePolynomialEvaluationResultDtoFactory;
21  import com.irurueta.reactorworkshop.polynomial.api.mappers.EvaluationStepMapper;
22  import com.irurueta.reactorworkshop.polynomial.api.mappers.PolynomialEvaluationResultMapper;
23  import com.irurueta.reactorworkshop.polynomial.domain.services.MultiplePolynomialEvaluationService;
24  import com.irurueta.reactorworkshop.polynomial.domain.services.PolynomialParametersValidationService;
25  import com.irurueta.reactorworkshop.polynomial.domain.services.ReactiveMultiplePolynomialEvaluationService;
26  import org.springframework.web.bind.annotation.PostMapping;
27  import org.springframework.web.bind.annotation.RequestBody;
28  import org.springframework.web.bind.annotation.RequestMapping;
29  import org.springframework.web.bind.annotation.RequestParam;
30  import org.springframework.web.bind.annotation.RestController;
31  import reactor.core.publisher.Flux;
32  import reactor.core.publisher.Mono;
33  
34  import java.time.Duration;
35  import java.time.Instant;
36  import java.util.stream.Collectors;
37  
38  /**
39   * REST controller in charge of evaluating polynomials, finding their roots, extrema, derivatives and integrations.
40   */
41  @RestController
42  @RequestMapping("/polynomial")
43  public class PolynomialController {
44  
45      /**
46       * Service in charge of non-reactively evaluating multiple trees of polynomial evaluation steps, obtaining their
47       * final results and finding their roots, extrema, derivatives and integrations.
48       */
49      private final MultiplePolynomialEvaluationService evaluationService;
50  
51      /**
52       * Service in charge of reactively evaluating multiple trees of polynomial evaluation steps, obtaining their final
53       * results and finding their roots, extrema, derivatives and integrations.
54       */
55      private final ReactiveMultiplePolynomialEvaluationService reactiveEvaluationService;
56  
57      /**
58       * Service to validate provided input polynomials.
59       */
60      private final PolynomialParametersValidationService validationService;
61  
62      /**
63       * Maps polynomial evaluation steps between DTO's and domain.
64       */
65      private final EvaluationStepMapper stepMapper;
66  
67      /**
68       * Maps polynomial evaluation results between domain and DTO's.
69       */
70      private final PolynomialEvaluationResultMapper resultMapper;
71  
72      /**
73       * Build DTO instances containing results of requests.
74       */
75      private final MultiplePolynomialEvaluationResultDtoFactory factory;
76  
77      /**
78       * Constructor.
79       *
80       * @param evaluationService         service in charge of non-reactively evaluating polynomials.
81       * @param reactiveEvaluationService service in charge of reactively evaluating polynomials.
82       * @param validationService         service to validate order of provided input polynomials.
83       * @param stepMapper                maps polynomial evaluation steps from DTO's.
84       * @param resultMapper              maps polynomial evaluation results to DTO's.
85       * @param factory                   builds DTO instances containing results.
86       */
87      public PolynomialController(
88              final MultiplePolynomialEvaluationService evaluationService,
89              final ReactiveMultiplePolynomialEvaluationService reactiveEvaluationService,
90              final PolynomialParametersValidationService validationService,
91              final EvaluationStepMapper stepMapper,
92              final PolynomialEvaluationResultMapper resultMapper,
93              final MultiplePolynomialEvaluationResultDtoFactory factory) {
94          this.evaluationService = evaluationService;
95          this.reactiveEvaluationService = reactiveEvaluationService;
96          this.validationService = validationService;
97          this.stepMapper = stepMapper;
98          this.resultMapper = resultMapper;
99          this.factory = factory;
100     }
101 
102     /**
103      * Evaluates a collection of evaluation steps to compute the final polynomials for each provided step and compute
104      * their roots, extrema, derivatives and integrations.
105      * This implementation is non reactive.
106      * Optionally a delay can be provided between evaluations of different polynomials to observe the impact in
107      * performance (used memory, cpu and threads) in the server.
108      *
109      * @param steps       a collection containing trees of evaluation steps to obtain polynomials.
110      * @param delayMillis optional amount of delay between polynomials in the collection (expressed in milliseconds).
111      * @return results of evaluation.
112      * @throws InterruptedException if thread is interrupted.
113      */
114     @PostMapping("/non-reactive")
115     public MultiplePolynomialEvaluationResultDto evaluateNonReactive(
116             @RequestBody final EvaluationStepsDto steps,
117             @RequestParam(value = "delay", required = false) final Long delayMillis) throws InterruptedException {
118         final var startInstant = Instant.now();
119 
120         final var stepList = stepMapper.mapFromDto(steps);
121         validationService.validate(stepList);
122 
123         final var results = evaluationService.evaluate(stepList, delayMillis)
124                 .stream().map(resultMapper::mapToDto).collect(Collectors.toList());
125 
126         final var endInstant = Instant.now();
127         final var duration = Duration.between(startInstant, endInstant);
128 
129         return factory.build(results, duration);
130     }
131 
132     /**
133      * Evaluates a collection of evaluation steps to compute the final polynomials for each provided step and compute
134      * their roots, extrema, derivatives and integrations.
135      * This implementation is reactive.
136      * Optionally a delay can be provided between evaluations of different polynomials to observe the impact in
137      * performance (used memory, cpu and threads) in the server.
138      *
139      * @param steps       a collection containing trees of evaluation steps to obtain polynomials.
140      * @param delayMillis optional amount of delay between polynomials in the collection (expressed in milliseconds).
141      * @return results of evaluation.
142      */
143     @PostMapping("/reactive")
144     public Mono<MultiplePolynomialEvaluationResultDto> evaluateReactive(
145             @RequestBody final EvaluationStepsDto steps,
146             @RequestParam(value = "delay", required = false) final Long delayMillis) {
147         var startInstant = Instant.now();
148 
149         final var stepList = stepMapper.mapFromDto(steps);
150         validationService.validate(stepList);
151 
152         return reactiveEvaluationService.evaluate(Flux.fromIterable(stepList), delayMillis)
153                 .map(resultMapper::mapToDto).collectList().map(results -> {
154                     final var endInstant = Instant.now();
155                     final var duration = Duration.between(startInstant, endInstant);
156 
157                     return factory.build(results, duration);
158                 });
159     }
160 }
