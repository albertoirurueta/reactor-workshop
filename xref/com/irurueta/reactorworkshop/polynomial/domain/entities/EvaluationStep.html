16  package com.irurueta.reactorworkshop.polynomial.domain.entities;
17  
18  import com.irurueta.numerical.polynomials.Polynomial;
19  import lombok.Data;
20  
21  /**
22   * Contains a step in the tree of evaluations to compute a polynomial.
23   */
24  @Data
25  public class EvaluationStep {
26  
27      /**
28       * Operation to perform at this step.
29       * {@link Operation#SUMMATION} is used to add two polynomials obtained from evaluating the steps in
30       * {@link #operand1} and {@link #operand2}.
31       * {@link Operation#SUBTRACTION} is used to subtract two polynomials obtained from evaluating the steps in
32       * {@link #operand1} and {@link #operand2}.
33       * {@link Operation#MULTIPLICATION} is used to multiply two polynomials obtained from evaluating the steps in
34       * {@link #operand1} and {@link #operand2}.
35       * {@link Operation#LITERAL} indicates that this step contains the literal polynomial parameters to be used.
36       */
37      private Operation operation;
38  
39      /**
40       * Contains an additional step to be used as 1st operand for {@link Operation#SUMMATION},
41       * {@link Operation#SUBTRACTION} or {@link Operation#MULTIPLICATION} operations.
42       */
43      private EvaluationStep operand1;
44  
45      /**
46       * Contains an additional step to be used as 2nd operand for {@link Operation#SUMMATION},
47       * {@link Operation#SUBTRACTION} or {@link Operation#MULTIPLICATION} operations.
48       */
49      private EvaluationStep operand2;
50  
51      /**
52       * Polynomial to be used literally.
53       */
54      private Polynomial literalPolynomial;
55  }
