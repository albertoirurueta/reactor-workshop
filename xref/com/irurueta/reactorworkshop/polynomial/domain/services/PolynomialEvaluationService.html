16  package com.irurueta.reactorworkshop.polynomial.domain.services;
17  
18  import com.irurueta.numerical.NumericalException;
19  import com.irurueta.numerical.polynomials.Polynomial;
20  import com.irurueta.reactorworkshop.polynomial.domain.entities.EvaluationStep;
21  import com.irurueta.reactorworkshop.polynomial.domain.entities.Operation;
22  import com.irurueta.reactorworkshop.polynomial.domain.entities.PolynomialEvaluationResult;
23  import com.irurueta.reactorworkshop.polynomial.domain.exceptions.InvalidEvaluationStepException;
24  import com.irurueta.reactorworkshop.polynomial.domain.exceptions.PolynomialEvaluationException;
25  import org.springframework.stereotype.Service;
26  
27  /**
28   * Evaluates a single evaluation step tree to obtain a single resulting polynomial and its associated results (roots,
29   * extrema, derivatives and integration).
30   */
31  @Service
32  public class PolynomialEvaluationService {
33  
34      /**
35       * Evaluates a single evaluation step tree to obtain a single resulting polynomial and its associated results
36       * (roots, extrema, derivatives and integration).
37       *
38       * @param step a single evaluation step tree.
39       * @return results for resulting polynomial.
40       * @throws PolynomialEvaluationException if computation of roots or extrema of resulting evaluated polynomial fails
41       *                                       for some reason.
42       * @throws InvalidEvaluationStepException if tree of evaluation steps contains invalid data.
43       */
44      public PolynomialEvaluationResult evaluate(final EvaluationStep step) {
45          try {
46              final var polynomial = buildPolynomialFromStep(step);
47  
48              final var roots = polynomial.getRoots();
49              final var minima = polynomial.getMinima();
50              final var maxima = polynomial.getMaxima();
51  
52              final var firstDerivative = polynomial.derivativeAndReturnNew();
53              final var secondDerivative = polynomial.nthDerivativeAndReturnNew(2);
54              final var integration = polynomial.integrationAndReturnNew();
55  
56              return PolynomialEvaluationResult.builder()
57                      .roots(roots)
58                      .minima(minima)
59                      .maxima(maxima)
60                      .polynomial(polynomial)
61                      .firstDerivative(firstDerivative)
62                      .secondDerivative(secondDerivative)
63                      .integration(integration)
64                      .build();
65  
66          } catch (final NumericalException e) {
67              throw new PolynomialEvaluationException(e);
68          }
69      }
70  
71      /**
72       * Internally evaluates a single step.
73       * If the step is literal, returns provided input, otherwise it is evaluated.
74       *
75       * @param step step to be evaluated.
76       * @return evaluated step or provided one if it is literal.
77       * @throws InvalidEvaluationStepException if step has no associated operation.
78       */
79      private EvaluationStep internalEvaluate(final EvaluationStep step) {
80          if (step.getOperation() != null) {
81              switch (step.getOperation()) {
82                  case SUMMATION:
83                      return evaluateSummationStep(step);
84                  case SUBTRACTION:
85                      return evaluateSubtractionStep(step);
86                  case MULTIPLICATION:
87                      return evaluateMultiplicationStep(step);
88                  case LITERAL:
89                      validateLiteralStep(step);
90                      return step;
91              }
92          }
93  
94          throw new InvalidEvaluationStepException("Invalid operation");
95      }
96  
97      /**
98       * Evaluates a step containing a summation of two polynomials.
99       *
100      * @param step step to be evaluated.
101      * @return a literal step containing summation of input step operands.
102      */
103     private EvaluationStep evaluateSummationStep(final EvaluationStep step) {
104         validateOperands(step);
105 
106         final var polynomial1 = buildPolynomialFromStep(step.getOperand1());
107         final var polynomial2 = buildPolynomialFromStep(step.getOperand2());
108 
109         return buildLiteralEvaluationStep(polynomial1.addAndReturnNew(polynomial2));
110     }
111 
112     /**
113      * Evaluates a step containing a subtraction of two polynomials.
114      *
115      * @param step step to be evaluated.
116      * @return a literal step containing subtraction of input step operands.
117      */
118     private EvaluationStep evaluateSubtractionStep(final EvaluationStep step) {
119         validateOperands(step);
120 
121         final var polynomial1 = buildPolynomialFromStep(step.getOperand1());
122         final var polynomial2 = buildPolynomialFromStep(step.getOperand2());
123 
124         polynomial2.multiplyByScalar(-1.0);
125 
126         return buildLiteralEvaluationStep(polynomial1.addAndReturnNew(polynomial2));
127     }
128 
129     /**
130      * Evaluates a step containing a multiplication of two polynomials.
131      *
132      * @param step step to be evaluated.
133      * @return a literal step containing multiplication of intput step operands.
134      */
135     private EvaluationStep evaluateMultiplicationStep(final EvaluationStep step) {
136         validateOperands(step);
137 
138         final var polynomial1 = buildPolynomialFromStep(step.getOperand1());
139         final var polynomial2 = buildPolynomialFromStep(step.getOperand2());
140 
141         return buildLiteralEvaluationStep(polynomial1.multiplyAndReturnNew(polynomial2));
142     }
143 
144     /**
145      * Builds a polynomial from an evaluation step by evaluating it and returning its literal polynomial.
146      *
147      * @param step step to be evaluated.
148      * @return evaluated polynomial.
149      */
150     private Polynomial buildPolynomialFromStep(final EvaluationStep step) {
151         return internalEvaluate(step).getLiteralPolynomial();
152     }
153 
154     /**
155      * Builds a literal evaluation step from provided polynomial.
156      *
157      * @param polynomial a polynomial to build a literal evaluation step.
158      * @return a literal evaluation step containing provided polynomial.
159      */
160     private EvaluationStep buildLiteralEvaluationStep(final Polynomial polynomial) {
161         final var step = new EvaluationStep();
162         step.setOperation(Operation.LITERAL);
163         step.setLiteralPolynomial(polynomial);
164 
165         return step;
166     }
167 
168     /**
169      * Validates operands of a non-literal evaluation step requiring both operands.
170      *
171      * @param step step to be validated.
172      * @throws InvalidEvaluationStepException if any operand is missing.
173      */
174     private void validateOperands(final EvaluationStep step) {
175         if (step.getOperand1() == null || step.getOperand2() == null) {
176             throw new InvalidEvaluationStepException("Operands are missing");
177         }
178     }
179 
180     /**
181      * Validates that a literal step contains a literal polynomial.
182      *
183      * @param step step to be validated.
184      * @throws InvalidEvaluationStepException if literal polynomial is missing.
185      */
186     private void validateLiteralStep(final EvaluationStep step) {
187         if (step.getLiteralPolynomial() == null) {
188             throw new InvalidEvaluationStepException("Literal polynomial is missing");
189         }
190     }
191 }
