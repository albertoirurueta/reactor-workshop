16  package com.irurueta.reactorworkshop;
17  
18  import java.lang.annotation.Annotation;
19  
20  /**
21   * Utility class for tests.
22   */
23  public class TestUtils {
24  
25      /**
26       * Gets annotation at class level.
27       *
28       * @param c               class to obtain an annotation from.
29       * @param annotationClass annotation class to be obtained.
30       * @param <T>             type of annotation to be obtained.
31       * @return found annotation or null.
32       */
33      public static <T extends Annotation> T getClassAnnotation(final Class<?> c, final Class<T> annotationClass) {
34          return c.getAnnotation(annotationClass);
35      }
36  
37      /**
38       * Gets annotation at method level.
39       *
40       * @param annotationClass annotation class to be obtained.
41       * @param object          object to get annotations from.
42       * @param methodName      name of method where annotation is searched.
43       * @param parameterTypes  parameter types of method where annotation is searched (can be ignored if method has no
44       *                        parameters).
45       * @param <T>             type of annotation to be obtained.
46       * @return found annotation or null.
47       * @throws NoSuchMethodException if method with provided name and parameter types does not exist on provided
48       *                               object class.
49       */
50      public static <T extends Annotation> T getMethodAnnotation(
51              final Class<T> annotationClass, final Object object, final String methodName,
52              final Class<?>... parameterTypes) throws NoSuchMethodException {
53          final var method = object.getClass().getMethod(methodName, parameterTypes);
54          return method.getDeclaredAnnotation(annotationClass);
55      }
56  
57      /**
58       * Gets annotations for parameters in a method.
59       *
60       * @param object         object to get annotations from.
61       * @param methodName     name of method where annotations are searched.
62       * @param parameterTypes parameter types of methods where annotations are searched (can be ignored if method has no
63       *                       parameters).
64       * @return array of array containing parameter annotations. First index indicates parameter position in the method,
65       * second index indicates annotation position for the method. First dimension length will always be the total number
66       * of parameters in the method. Null values indicate that a given parameter has no annotations.
67       * @throws NoSuchMethodException if method with provided name and parameter types does not exist on provided object
68       *                               class.
69       */
70      public static Annotation[][] getMethodParameterAnnotations(
71              final Object object, final String methodName, final Class<?>... parameterTypes)
72              throws NoSuchMethodException {
73          final var method = object.getClass().getDeclaredMethod(methodName, parameterTypes);
74          return method.getParameterAnnotations();
75      }
76  }
